int NTCN(int a,int b)
{
    int r;
    while(b)
        r = a % b, a = b, b = r;
    return (a == 1);
}
//Hàm này là dùng đệ quy
int NTCNDQ(int a,int b)
{
    if(b == 0)
        return (a == 1);
    return NTCNDQ(b,a%b);
}
